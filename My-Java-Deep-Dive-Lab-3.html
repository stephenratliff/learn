1. Modern Foundations

Modern JavaScript relies on block-scoped declarations, expressive destructuring, and safe property access. Understanding the Temporal Dead Zone (TDZ) prevents ReferenceError surprises when let/const are accessed before initialization. Destructuring paired with rest/spread yields immutable-friendly updates without manual copying loops.

Why it matters: TDZ and const-correctness eliminate a whole class of hoisting bugs. Optional chaining ?. and nullish coalescing ?? let you traverse uncertain API shapes without verbose guard clauses, keeping logic readable and CPU-friendly by avoiding unnecessary object allocations.

2. Functions Deep Dive

Functions are first-class values. Declarations are hoisted with their bodies; expressions are not. Arrow functions capture lexical this, making them ideal for callbacks but unsuitable for methods needing dynamic context. Closures retain outer variables, enabling private state without classes.

Why it matters: Proper closure usage avoids global mutable state. Currying transforms multi-argument functions into composable unary pipelines, improving testability and partial application in event handlers and middleware.
function createCounter(start = 0) {
  let count = start; // private
  return {
    inc: () => ++count,
    dec: () => --count,
    value: () => count,
    reset: () => { count = start; }
  };
}
const c = createCounter(5);
c.inc(); c.inc();
console.log('counter:', c.value()); // 7
const obj = {
  value: 10,
  regular() { return this.value; },
  arrow: () => this.value, // lexical -> undefined in module/global
};
console.log('regular:', obj.regular()); // 10
console.log('arrow:', obj.arrow()); // undefined
const bound = obj.regular.bind({ value: 99 });
console.log('bound:', bound()); // 99
const curry = fn => (...a) =>
  a.length >= fn.length ? fn(...a) : (...b) => curry(fn)(...a, ...b);

const add3 = (a,b,c) => a + b + c;
const curried = curry(add3);
const add5 = curried(2)(3); // returns function expecting c
console.log('curried result:', add5(10)); // 15

const partial = (fn, ...preset) => (...later) => fn(...preset, ...later);

3. Objects, Prototypes, Classes

Every object links to a prototype. Class syntax is sugar over constructor functions with prototype methods. Private fields (#) are truly private at runtime, unlike convention-based underscores.

Why it matters: Mastering the prototype chain explains property lookup performance and enables memory-efficient shared methods. Iterators (Symbol.iterator) make custom collections work with for...of and spread.
const animal = {
  speak() { return `${this.name} makes a noise`; }
};
const dog = Object.create(animal);
dog.name = 'Rex';
dog.speak = function(){ return `${animal.speak.call(this)}: woof`; };
console.log(dog.speak());
console.log('prototype is animal:', Object.getPrototypeOf(dog) === animal);
class Account {
  #balance = 0; // private
  constructor(owner){ this.owner = owner; }
  deposit(amount){ if(amount<=0) throw new Error('bad amount'); this.#balance += amount; }
  withdraw(amount){ if(amount > this.#balance) throw new Error('insufficient'); this.#balance -= amount; }
  get balance(){ return this.#balance; }
  static from(obj){ const a = new Account(obj.owner); a.deposit(obj.balance||0); return a; }
  *[Symbol.iterator](){ yield this.owner; yield this.#balance; }
}

4. Asynchronous JavaScript

The event loop processes tasks in phases. Microtasks (Promise reactions, queueMicrotask) run after each macrotask (setTimeout, I/O). This ordering explains why promises resolve before timers.

Why it matters: Correct async composition avoids race conditions. AbortController provides cooperative cancellation for fetch and other async operations, preventing wasted bandwidth and memory leaks.
console.log('start');
setTimeout(() => console.log('macrotask: timeout'), 0);
Promise.resolve().then(() => console.log('microtask: promise'));
queueMicrotask(() => console.log('microtask: queueMicrotask'));
console.log('end');
async function fetchWithTimeout(url, ms = 3000){
  const ctrl = new AbortController();
  const id = setTimeout(() => ctrl.abort(), ms);
  try{
    const res = await fetch(url, { signal: ctrl.signal });
    if(!res.ok) throw new Error('HTTP ' + res.status);
    return await res.json();
  } finally { clearTimeout(id); }
}
// fetchWithTimeout('https://jsonplaceholder.typicode.com/todos/1').then(console.log).catch(console.error);

5. Fetch, APIs, and Modules

Wrap fetch with consistent error handling returning a tuple [data, error]. Dynamic import enables code-splitting on interaction.

Why it matters: Centralizing fetch logic ensures timeouts, JSON parsing errors, and HTTP errors are handled uniformly across your app.
export async function safeFetch(url, options = {}){
  try{
    const res = await fetch(url, options);
    if(!res.ok) throw new Error(`HTTP ${res.status}`);
    const data = await res.json();
    return [data, null];
  }catch(err){
    return [null, err];
  }
}

7. Web Storage and State

localStorage stores strings only—wrap with JSON helpers. IndexedDB provides transactional, asynchronous storage for structured data exceeding 5MB.

Why it matters: Proper serialization prevents data corruption. IndexedDB version upgrades must create stores in onupgradeneeded to avoid InvalidState errors.
const DB_NAME = 'handbook-notes';
const STORE = 'notes';
function openDB(){
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, 1);
    req.onupgradeneeded = () => req.result.createObjectStore(STORE, { keyPath: 'id' });
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}
export async function saveNote(note){
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE, 'readwrite');
    tx.objectStore(STORE).put(note);
    tx.oncomplete = () => resolve(true);
    tx.onerror = () => reject(tx.error);
  });
}

9. Node.js Essentials

Node provides fs/promises for non-blocking file I/O, path for cross-platform paths, and a built-in HTTP module sufficient for tiny static servers.

Why it matters: Watching directories and prettifying JSON prevents drift in config files. A minimal static server accelerates prototyping without external dependencies.
import { watch } from 'node:fs';
import { readFile, writeFile } from 'node:fs/promises';

watch('./data', async (event, filename) => {
  if (!filename?.endsWith('.json')) return;
  const path = `./data/${filename}`;
  try{
    const raw = await readFile(path, 'utf8');
    const pretty = JSON.stringify(JSON.parse(raw), null, 2);
    await writeFile(path, pretty);
    console.log('Prettified', filename);
  }catch(e){ console.error('Failed', filename, e.message); }
});
import http from 'node:http';
import { readFile } from 'node:fs/promises';
import path from 'node:path';

const MIME = { '.html':'text/html', '.js':'text/javascript', '.css':'text/css', '.json':'application/json' };

const server = http.createServer(async (req, res) => {
  const urlPath = req.url === '/' ? '/index.html' : req.url;
  const filePath = path.join(process.cwd(), 'public', path.normalize(urlPath));
  try {
    const data = await readFile(filePath);
    const ext = path.extname(filePath);
    res.writeHead(200, { 'Content-Type': MIME[ext] || 'application/octet-stream' });
    res.end(data);
  } catch {
    res.writeHead(404); res.end('Not found');
  }
});
server.listen(8080, () => console.log('http://localhost:8080'));

11. Performance and Debugging

Debounce coalesces rapid events; throttle guarantees max frequency. Memoize caches pure function results keyed by arguments.

export function debounce(fn, wait=200){
  let t; return (...args) => { clearTimeout(t); t = setTimeout(()=>fn(...args), wait); };
}
export function throttle(fn, limit=200){
  let last=0; return (...args) => {
    const now = Date.now();
    if(now - last >= limit){ last = now; fn(...args); }
  };
}
export function memoize(fn){
  const cache = new Map();
  return (...args) => {
    const key = JSON.stringify(args);
    if(cache.has(key)) return cache.get(key);
    const val = fn(...args);
    cache.set(key, val);
    return val;
  };
}

  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), timeout);
  try{
    const res = await fetch('https://jsonplaceholder.typicode.com/todos/' + id, { signal: ctrl.signal });
    if(!res.ok) throw new Error(`HTTP ${res.status}`);
